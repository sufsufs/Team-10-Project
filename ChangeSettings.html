    <script>

        // Add event listener for form submission

        document.getElementById('profileForm').addEventListener('submit', function(event) {

            event.preventDefault(); // Prevent default form submission

            // Get values from input fields

            const username = document.getElementById('username').value;

            const email = document.getElementById('email').value;

            const password = document.getElementById('password').value;

            const language = document.getElementById('language').value;

            let errorMessages = []; // Array to hold error messages

            // Validate username

            if (username.length < 3) {

                errorMessages.push('Username must be at least 3 characters long.');

            }

            // Validate email using regex pattern

            const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;

            if (!emailPattern.test(email)) {

                errorMessages.push('Please enter a valid email address.');

            }

            // Validate password

            if (password.length < 6) {

                errorMessages.push('Password must be at least 6 characters long.');

            }

            // Validate language selection

            if (!language) {

                errorMessages.push('Please select a preferred language.');

            }

            // Get the error container div

            const errorContainer = document.getElementById('errorContainer');

            errorContainer.innerHTML = ''; // Clear previous error messages

            // If there are validation errors, display them

            if (errorMessages.length > 0) {

                errorContainer.innerHTML = errorMessages.join('<br>');

            } else {

                // If no errors, create user data object and log it

                const userData = {

                    username: username,

                    email: email,

                    password: password,

                    language: language

                };

                console.log('User Data:', userData);

                alert('Profile updated successfully!'); // Show success message

            }

        });
